 31 /**
 32  * Runs supplied Minima SQL on given table
 33  * @function doSQL
 34  * @param {string} sql
 35  * @param {string} tableName
 36 */
 37 function doSQL(sql, tableName) {
 38   Minima.sql(sql, function(resp) {
 39     if (!resp.status) {
 40       Minima.log(app + ' Error with SQL ' + tableName + resp.message);
 41     } else {
 42       Minima.log(app + ' SQL ' + tableName);
 43     }
 44   });
 45 }
 46 
 47 /**
 48  * Creates log entries
 49  * @function doLog
 50  * @param {string} typeId
 51  * @param {string} type
 52  * @param {string} data
 53 */
 54 function doLog(typeId, type, data) {
 55   const tableName = tables.log.name;
 56   const date = Date.now();
 57   const insertSQL = 'INSERT INTO ' +
 58       tableName +
 59       ' (loggingTypeId, loggingType, date, data) ' +
 60       'VALUES (' +
 61       '\'' + typeId + '\', ' +
 62       '\'' + type + '\', ' +
 63       '\'' + date + '\', ' +
 64       '\'' + data + '\'' +
 65     ')';
 66   doSQL(insertSQL, tableName);
 67 }

 69 /**
 70  * Creates TxPow table
 71  * @function createTxPow
 72  */
 73 function createTxPow() {

 75   const createSQL = 'CREATE Table IF NOT EXISTS ' +
 76       tableName + ' (' +
 77       'id varchar(255) NOT NULL, ' +
 78       'PRIMARY KEY(id)' +
 79     ');';
 80   doSQL(createSQL, tableName);
 81 }

120 /**

122  * @function createLog
123  */
124 function createLog() {
125   const tableName = tables.log.name;
126   const createSQL = 'CREATE Table IF NOT EXISTS ' +
127       tableName + ' (' +

139 /**
140  * Creates URL table
141  * @function createURL
142  */
143 function createURL() {
144   const tableName = tables.url.name;
145   const createSQL = 'CREATE Table IF NOT EXISTS ' +
146       tableName + ' (' +
147       'name varchar(255) NOT NULL, ' +
148       'triggername varchar(255) NOT NULL, ' +
149       'PRIMARY KEY(name), ' +
150       'FOREIGN KEY (triggername) REFERENCES trigger(name)' +

152     'CREATE INDEX IF NOT EXISTS arrange_index ON ' +

166 /**
167  * Adds relevant txpows to TxPow table
168  * @function addTxPoW
169  * @param {object} txpow
170 */
171 function addTxPoW(txpow) {
172   const txOutputs = txpow.body.txn.outputs;
173   if ( ( Array.isArray(txOutputs) ) &&
174         ( txOutputs.length ) ) {
175     const id = txpow.txpowid;
176     const tableName = tables.txpow.name;
177     const insertSQL = 'INSERT INTO ' +
178       tableName +
179       ' (id) ' +
180       'VALUES (' +
181       '\'' + id + '\'' +
182     ')';
183 
184     doSQL(insertSQL, tableName);
185     doLog(id, logTypes.TXPOW, 'insert');
186   }
187 }

193   } else if (msg.event == 'newtxpow') {
194     addTxPoW(msg.info.txpow);
195   }
